SELECT     dbo.Налоговый_агент.ID, dbo.Налоговый_агент.DEL, dbo.Налоговый_агент.ID_платежный_док, 
                      dbo.Налоговый_агент.Наимен_организации AS [Наименование организации], dbo.Налоговый_агент.Страна, 
                      dbo.Налоговый_агент.Вид_дохода AS [Вид дохода], dbo.Налоговый_агент.Дата_платежа AS [Дата начисления платежа], 
                      dbo.Налоговый_агент.Сумма_платежа AS [Сумма платежа], 
                      dbo.Налоговый_агент.Сумма_затрат_для_налога AS [Сумма затрат для исчисления налога], 
                      dbo.Налоговый_агент.Сумма_освоб_от_налога_РБ AS [Сумма дохода осв от налога по зак РБ], 
                      dbo.Налоговый_агент.Сумма_освоб_от_налога_международн AS [по международному договору], 
                      dbo.Налоговый_агент.Сумма_платежа - ISNULL(dbo.Налоговый_агент.Сумма_затрат_для_налога, 0) 
                      - ISNULL(dbo.Налоговый_агент.Сумма_освоб_от_налога_РБ, 0) - ISNULL(dbo.Налоговый_агент.Сумма_освоб_от_налога_международн, 0) 
                      AS [облагаемый доход], dbo.Налоговый_агент.Ставка_налога_РБ AS [Ставка налога по зак РБ], 
                      dbo.Налоговый_агент.Ставка_налога_международн AS [ставка по международному договору], 
                      (dbo.Налоговый_агент.Сумма_платежа - ISNULL(dbo.Налоговый_агент.Сумма_затрат_для_налога, 0) 
                      - ISNULL(dbo.Налоговый_агент.Сумма_освоб_от_налога_РБ, 0) - ISNULL(dbo.Налоговый_агент.Сумма_освоб_от_налога_международн, 0)) 
                      * (CAST(ISNULL(dbo.Налоговый_агент.Ставка_налога_РБ, 0) + ISNULL(dbo.Налоговый_агент.Ставка_налога_международн, 0) AS REAL) / 100) 
                      AS [Подлежит уплате], dbo.view_Платежный_док.Сумма AS Перечислено, dbo.view_Платежный_док.Дата AS [Дата плат инстр], 
                      dbo.view_Платежный_док.Номер_платежного_документа AS [Номер плат инстр]
FROM         dbo.Налоговый_агент LEFT OUTER JOIN
                      dbo.view_Платежный_док ON dbo.Налоговый_агент.ID_платежный_док = dbo.view_Платежный_док.ID_платежный_док
WHERE     (dbo.Налоговый_агент.DEL = 0)
