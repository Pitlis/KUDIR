SELECT     ID, DEL, Уплачено_при_приобретении, Док_приобр_поставщ_Дата AS [Дата документа поставщика], 
                      Док_приобр_поставщ_Номер AS [Номер документа поставщика], Наим_прод_покуп AS [Наименование покупателя], 
                      Уч_номер_плательщ AS [Учетный номер плательщика], Дата_оплаты AS [Дата оплаты], НДС AS [Размер НДС], 
                      Стоим_с_НДС AS [Стоимость включая НДС], CAST(Стоим_с_НДС AS REAL) / (1 + CAST(НДС AS REAL) / 100) AS [Стоимость без НДС], 
                      (CAST(Стоим_с_НДС AS REAL) / (1 + CAST(НДС AS REAL) / 100)) * (CAST(НДС AS REAL) / 100) AS [Сумма НДС], 
                      Стоим_освоб_от_НДС_всего AS [Стоимость объектов освобождаемых от НДС], 
                      Оборот_за_пределами_РБ AS [из них оборот за пределами РБ]
FROM         dbo.НДС
WHERE     (DEL = 0) AND (Уплачено_при_приобретении = 0)
