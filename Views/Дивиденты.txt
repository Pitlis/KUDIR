SELECT     dbo.Налоги_на_дивиденты.ID, dbo.view_Платежный_док.ID_платежный_док, 
                      dbo.Налоги_на_дивиденты.Наименование_организации AS [Наименование организации], 
                      dbo.Налоги_на_дивиденты.Дата_начисления_дивидентов AS [Дата начисления], dbo.Налоги_на_дивиденты.Сумма, 
                      dbo.Налоги_на_дивиденты.Налоговая_база AS [Налоговая база], dbo.Налоги_на_дивиденты.Ставка_налога AS [Ставка налога], 
                      dbo.Налоги_на_дивиденты.Налоговая_база * (dbo.Налоги_на_дивиденты.Ставка_налога / 100) AS [Сумма налога], 
                      dbo.view_Платежный_док.Дата AS [Дата плат инстр], dbo.view_Платежный_док.Номер_платежного_документа AS [Номер плат инстр], 
                      dbo.view_Платежный_док.Сумма AS [Перечислено deналога], dbo.Налоги_на_дивиденты.DEL
FROM         dbo.Налоги_на_дивиденты LEFT OUTER JOIN
                      dbo.view_Платежный_док ON dbo.Налоги_на_дивиденты.ID_платежный_док = dbo.view_Платежный_док.ID_платежный_док
WHERE     (dbo.Налоги_на_дивиденты.DEL = 0)
