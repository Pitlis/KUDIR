SELECT     ID, Дата_отгрузки_предоплаты AS [Дата предоплаты], Лицо_которому_реализуется_товар AS [Лицо которому реализ товар], 
                      Док_отгр_опл_Наим AS [Наим док оплаты], Док_отгр_опл_Номер AS [Номер док оплаты], Док_отгр_опл_Дата AS [Дата док оплаты], 
                      Стоимость_в_бр AS [Сумма предоплаты], Наименование_валюты AS [Наим валюты], Стоимость_в_валюте AS [Сумма в валюте], 
                      Док_опл_отгр_Наим AS [Наим док отгрузки], Док_опл_отгр_Номер AS [Номер док отгрузки], Док_опл_отгр_Дата AS [Дата док отгрузки], 
                      Предоплачено, DEL
FROM         dbo.Отгрузка_товаров
WHERE     (Предоплачено = 1) AND (DEL = 0)
