SELECT     dbo.Выплаты.Код_выплаты, dbo.Выплата_работнику.работникID, dbo.Выплаты.DEL AS DEL_Выплаты, dbo.Выплаты.Сумма, 
                      dbo.Выплаты.Причина, dbo.Выплаты.Период_Начало AS [Начало периода], dbo.Выплаты.Период_Конец AS [Конец периода], 
                      dbo.Выплаты.Начисляются_страх_взносы AS [Начисляются страх взносы], 
                      dbo.Выплаты.Начисляются_пенс_взносы AS [Начисляются пенс взносы], dbo.Выплата_работнику.DEL AS DEL_ВыплатыРаботнику
FROM         dbo.Выплаты INNER JOIN
                      dbo.Выплата_работнику ON dbo.Выплаты.Код_выплаты = dbo.Выплата_работнику.Код_выплаты
WHERE     (dbo.Выплаты.DEL = 0) AND (dbo.Выплата_работнику.DEL = 0)
