SELECT     ID, Дата_отгрузки_предоплаты AS [Дата отгрузки], Лицо_которому_реализуется_товар AS [Лицо которому реализ товар], 
                      Док_отгр_опл_Наим AS [Наим док отгрузки], Док_отгр_опл_Номер AS [Номер док отгрузки], Док_отгр_опл_Дата AS [Дата док отгрузки], 
                      Стоимость_в_бр AS [Стоимость отгр товаров в бр], Наименование_валюты AS [Наим валюты], Стоимость_в_валюте AS [Стоимость в валюте], 
                      Док_опл_отгр_Наим AS [Наим док оплаты], Док_опл_отгр_Номер AS [Номер док оплаты], Док_опл_отгр_Дата AS [Дата док оплаты], 
                      Сумма_оплаты_отгр AS Сумма, DEL, Предоплачено, Дата_оплаты_отгр AS [Дата оплаты]
FROM         dbo.Отгрузка_товаров
WHERE     (DEL = 0) AND (Предоплачено = 0)
