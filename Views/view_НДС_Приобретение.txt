SELECT     ID, DEL, Уплачено_при_приобретении, Док_приобр_поставщ_Дата AS [Дата приобретения], 
                      Док_приобр_поставщ_Номер AS [Номер документа приобр], Дата_оплаты AS [Дата оплаты], Номер_док_оплаты AS [Номер документа оплаты], 
                      Наим_прод_покуп AS [Наименование продавца], Уч_номер_плательщ AS [Учетный номер плательщика], НДС AS [Размер НДС], 
                      Стоим_с_НДС AS [Стоимость включая НДС], CAST(Стоим_с_НДС AS REAL) / (1 + CAST(НДС AS REAL) / 100) AS [Стоимость без НДС], 
                      (CAST(Стоим_с_НДС AS REAL) / (1 + CAST(НДС AS REAL) / 100)) * (CAST(НДС AS REAL) / 100) AS [Сумма НДС]
FROM         dbo.НДС
WHERE     (DEL = 0) AND (Уплачено_при_приобретении = 1)
