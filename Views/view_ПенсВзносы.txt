SELECT     dbo.Работник.работникID, dbo.Работник.ФИО, dbo.Пенсионный_взнос.Дата, 
                      dbo.Get_СуммаВыплатНачисляютсяПенсВзносы(dbo.Работник.работникID, dbo.Пенсионный_взнос.Дата) 
                      AS [Сумма на которую начисл пенс взносы], dbo.Get_СуммаВыплатНачисляютсяПенсВзносы(dbo.Работник.работникID, 
                      dbo.Пенсионный_взнос.Дата) * (CAST(dbo.Работник.Тариф_пенс_взносов AS REAL) / 100) AS [Сумма начисленных пенс взносов], 
                      dbo.Пенсионный_взнос.Задолж_за_пред_период AS [Остаток задолженности за пред период], 
                      dbo.Пенсионный_взнос.Иные_платежи AS [Иные платежи], dbo.Get_СуммаВыплатНачисляютсяПенсВзносы(dbo.Работник.работникID, 
                      dbo.Пенсионный_взнос.Дата) * (CAST(dbo.Работник.Тариф_пенс_взносов AS REAL) / 100) 
                      + ISNULL(dbo.Пенсионный_взнос.Задолж_за_пред_период, 0) + ISNULL(dbo.Пенсионный_взнос.Иные_платежи, 0) AS [Подлежит уплате], 
                      dbo.view_Платежный_док.Сумма AS [Перечислено в Фонд], dbo.view_Платежный_док.Дата AS [Дата плат инстр], 
                      dbo.view_Платежный_док.Номер_платежного_документа AS [Номер плат инстр], dbo.view_Платежный_док.ID_платежный_док, 
                      dbo.Пенсионный_взнос.DEL
FROM         dbo.Работник INNER JOIN
                      dbo.Пенсионный_взнос ON dbo.Работник.работникID = dbo.Пенсионный_взнос.работникID LEFT OUTER JOIN
                      dbo.view_Платежный_док ON dbo.Пенсионный_взнос.ID_платежный_док = dbo.view_Платежный_док.ID_платежный_док
WHERE     (dbo.Пенсионный_взнос.DEL = 0)
