SELECT     dbo.Налоговые_вычеты.Код_вычета, dbo.Налоговый_вычет_работника.работникID, dbo.Налоговые_вычеты.DEL AS DEL_вычет, 
                      dbo.Налоговые_вычеты.Сумма, dbo.Налоговые_вычеты.Тип_вычета AS [Тип вычета], 
                      dbo.Налоговые_вычеты.Период_Начало AS [Начало периода], dbo.Налоговые_вычеты.Период_Конец AS [Конец периода], 
                      dbo.Налоговый_вычет_работника.DEL AS DEL_вычетРаботника
FROM         dbo.Налоговый_вычет_работника INNER JOIN
                      dbo.Налоговые_вычеты ON dbo.Налоговый_вычет_работника.Код_вычета = dbo.Налоговые_вычеты.Код_вычета
WHERE     (dbo.Налоговые_вычеты.DEL = 0) AND (dbo.Налоговый_вычет_работника.DEL = 0)
