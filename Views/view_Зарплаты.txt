SELECT     dbo.Работник.ФИО, dbo.Выплаты.Сумма AS [Зарплата сумма], dbo.Выплаты.Период_Начало AS [Начало периода], 
                      dbo.Выплаты.Период_Конец AS [Конец периода]
FROM         dbo.Выплаты INNER JOIN
                      dbo.Выплата_работнику ON dbo.Выплаты.Код_выплаты = dbo.Выплата_работнику.Код_выплаты INNER JOIN
                      dbo.Работник ON dbo.Выплата_работнику.работникID = dbo.Работник.работникID
WHERE     (UPPER(dbo.Выплаты.Причина) LIKE UPPER('%Зарплат%')) AND (dbo.Выплаты.DEL = 0) AND (dbo.Выплата_работнику.DEL = 0) AND 
                      (dbo.Работник.DEL = 0)
