SELECT     dbo.Подоходный_налог.работникID, dbo.Подоходный_налог.DEL, dbo.view_Платежный_док.DEL AS Expr1, 
                      dbo.Подоходный_налог.Дата AS Месяц, dbo.Подоходный_налог.Начислено, dbo.view_Платежный_док.Сумма AS Перечислено, 
                      dbo.view_Платежный_док.Дата, dbo.view_Платежный_док.Номер_платежного_документа AS [Номер платежной инструкции]
FROM         dbo.Подоходный_налог INNER JOIN
                      dbo.view_Платежный_док ON dbo.Подоходный_налог.Дата = dbo.view_Платежный_док.Дата
WHERE     (dbo.view_Платежный_док.DEL = 0) AND (dbo.Подоходный_налог.DEL = 0)
