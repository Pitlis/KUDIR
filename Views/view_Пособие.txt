SELECT     dbo.Пособия.Код_пособия, dbo.Пособие_работника.работникID, dbo.Пособие_работника.DEL AS DEL_пособие, dbo.Пособия.Сумма, 
                      dbo.Пособия.Причина, dbo.Пособия.Период_Начало AS [Начало периода], dbo.Пособия.Период_Конец AS [Конец периода], 
                      dbo.Пособия.DEL AS DEl_пособиеРаботника
FROM         dbo.Пособие_работника INNER JOIN
                      dbo.Пособия ON dbo.Пособие_работника.Код_пособия = dbo.Пособия.Код_пособия
WHERE     (dbo.Пособие_работника.DEL = 0) AND (dbo.Пособия.DEL = 0)
