CREATE PROCEDURE upd_Выплата
@Код_выплаты INT,
@Причина VARCHAR(150),
@Сумма MONEY,
@Период_Начало DATETIME,
@Период_Конец DATETIME,
@Начисляются_страх_взносы BIT,
@Начисляются_пенс_взносы BIT
AS

IF (@Сумма IS NOT NULL)
	BEGIN
		UPDATE Выплаты SET Сумма = @Сумма, Причина = @Причина, Период_Начало = @Период_Начало, Период_Конец = @Период_Конец,
		Начисляются_страх_взносы = @Начисляются_страх_взносы, Начисляются_пенс_взносы = @Начисляются_пенс_взносы
		WHERE Код_выплаты = @Код_выплаты
	END
