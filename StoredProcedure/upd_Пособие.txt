CREATE PROCEDURE upd_Пособие
@Код_пособия INT,
@Причина VARCHAR(150),
@Сумма MONEY,
@Период_Начало DATETIME,
@Период_Конец DATETIME
AS

IF (@Сумма IS NOT NULL)
	BEGIN
		UPDATE Пособия SET Сумма = @Сумма, Причина = @Причина, Период_Начало = @Период_Начало, Период_Конец = @Период_Конец
		WHERE Код_пособия = @Код_пособия
	END
