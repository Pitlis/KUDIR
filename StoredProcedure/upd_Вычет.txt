CREATE PROCEDURE upd_Вычет
@Код_вычета INT,
@Тип_вычета VARCHAR(150),
@Сумма MONEY,
@Период_Начало DATETIME,
@Период_Конец DATETIME
AS

IF (@Сумма IS NOT NULL)
	BEGIN
		UPDATE Налоговые_вычеты SET Сумма = @Сумма, Тип_вычета = @Тип_вычета, Период_Начало = @Период_Начало, Период_Конец = @Период_Конец
		WHERE Код_вычета = @Код_вычета
	END
