CREATE PROCEDURE add_Пособие
@Причина VARCHAR(150),
@Сумма MONEY,
@Период_Начало DATETIME,
@Период_Конец DATETIME,
@IDRESULT INT OUTPUT
AS

INSERT INTO Пособия (Причина, Сумма, Период_Начало,	Период_Конец, DEL)
VALUES (@Причина, @Сумма, @Период_Начало, @Период_Конец, 0)
SET @IDRESULT = SCOPE_IDENTITY()
