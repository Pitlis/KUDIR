CREATE PROCEDURE add_ВыплатаРаботнику
@Причина VARCHAR(150),
@Сумма MONEY,
@Период_Начало DATETIME,
@Период_Конец DATETIME,
@Начисляются_страх_взносы BIT,
@Начисляются_пенс_взносы BIT,
@работникID INT
AS

DECLARE @IDdoc INT
IF (@Сумма IS NOT NULL)
	BEGIN
		EXEC add_Выплата @Причина, @Сумма, @Период_Начало, @Период_Конец, @Начисляются_страх_взносы, @Начисляются_пенс_взносы, @IDdoc output;
	END

INSERT INTO Выплата_работнику (работникID, Код_выплаты, DEL) VALUES (@работникID, @IDdoc, 0)
