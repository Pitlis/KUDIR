CREATE PROCEDURE add_Выплата
@Причина VARCHAR(150),
@Сумма MONEY,
@Период_Начало DATETIME,
@Период_Конец DATETIME,
@Начисляются_страх_взносы BIT,
@Начисляются_пенс_взносы BIT,
@IDRESULT INT OUTPUT
AS

INSERT INTO Выплаты (Причина, Сумма, Период_Начало,	Период_Конец, Начисляются_страх_взносы, Начисляются_пенс_взносы, DEL)
VALUES (@Причина, @Сумма, @Период_Начало, @Период_Конец, @Начисляются_страх_взносы, @Начисляются_пенс_взносы, 0)
SET @IDRESULT = SCOPE_IDENTITY()
