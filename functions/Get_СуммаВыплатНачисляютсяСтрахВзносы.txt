CREATE FUNCTION Get_СуммаВыплатНачисляютсяСтрахВзносы
 (@работникID INT, @Дата DATETIME)
RETURNS MONEY

BEGIN
	DECLARE @Summ MONEY
	
	SET @Summ = (SELECT SUM(ISNULL(Сумма,0)) 
	FROM view_Выплаты
	WHERE (@работникID = работникID) AND ([Начало периода] < @Дата) AND ([Конец периода] > @Дата) AND ([Начисляются страх взносы] = 1)
	)

	RETURN @Summ
END
